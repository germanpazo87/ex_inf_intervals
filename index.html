async function askGemini() { const qInput = document.getElementById('user-query'); const q = qInput.value; if (!q) return; addMessage(q, 'user'); qInput.value = ''; // CONTEXT PER A L'ALUMNE const context = `Ets el Core de la Matriu. NO donis solucions numèriques, dóna pistes curtes en to Cyberpunk.`; try { // ENDPOINT ACTUALITZAT A V1 (ESTABLE) const url = `https://generativelanguage.googleapis.com/v1/models/gemini-1.5-flash:generateContent?key=${API_KEY}`; const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: context + "\nPregunta de l'alumne: " + q }] }] }) }); const data = await response.json(); if (data.error) { // Això ens dirà exactament què passa (clau mal posada, quota, etc.) throw new Error(data.error.message); } const text = data.candidates[0].content.parts[0].text; addMessage(text, 'ia'); } catch (error) { addMessage("CRITICAL_ERROR: " + error.message, 'ia'); console.error("Detall del error:", error); } }
